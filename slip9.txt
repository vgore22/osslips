#String_explode.php


<?php
$str = $_POST['str'];
$sep = $_POST['sep'];

// (a) Split into words

$parts = explode($sep, $str);
echo "a) Words after splitting:<br>";
foreach ($parts as $w) {
    echo $w . "<br>";
}

// (b) Replace occurrences of separator with another (say @)

$newStr = str_replace($sep, "@", $str);
echo "<br>b) String after replacing '$sep' with '@': $newStr<br>";

// (c) Find the last word

$lastWord = end($parts);
echo "<br>c) Last word in string: $lastWord";
?>
