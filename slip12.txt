#functions.php


<?php
// Function with default parameters
function calculate($a, $b = 1, $op = "add") {
    if ($op == "add") {
        return $a + $b;
    }
    else if ($op == "sub") {
        return $a - $b;
    }
    else if ($op == "mul") {
        return $a * $b;
    }
    else if ($op == "div") {
        if ($b != 0)
            return $a / $b;
        else
            return "Division by zero error";
    }
    else {
        return "Invalid operation";
    }
}
?>
