# marksresult.php

<?php
$serial = explode(",", $_POST['serial']);
$subjects = explode(",",
$_POST['subjects']);
$marks = explode(",", $_POST['marks']);
$total = 0;
echo "<table border='1' cellpadding='5'
cellspacing='0'>";
echo "<tr><th>Serial No
</th><th>Subject</th><th>
Marks (out of 100)</th></tr>";
for ($i=0; $i<5; $i++) {
$sno = trim($serial[$i]);
$sub = trim($subjects[$i]);
$m = trim($marks[$i]);
$total += $m;
echo "<tr><td>$sno</td>
<td>$sub</td>
<td>$m</td></tr>";
}
echo "</table><br>";
$percentage = $total / 5;
if ($percentage >= 75) $grade =
"Distinction";
elseif ($percentage >= 60) $grade = "First
Class";
elseif ($percentage >= 50) $grade = "Second
Class";
elseif ($percentage >= 35) $grade = "Pass";
else $grade = "Fail";
echo "Total Marks = $total<br>";
echo "Percentage = $percentage %<br>";
echo "Grade = $grade<br>";
?>
