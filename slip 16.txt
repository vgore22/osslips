Slip16_1: Write a program to find the Square of given number using function interface.





import java.util.Scanner;


@FunctionalInterface
interface SquareFunction {
    int square(int n);
}

public class SquareUsingInterface {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);

        
        System.out.print("Enter a number: ");
        int num = sc.nextInt();

        
        SquareFunction sq = (n) -> n * n;

         int result = sq.square(num);

       
        System.out.println("Square of " + num + " is: " + result);

        sc.close();
    }
}
