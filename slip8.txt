#Stringpos.php

	<?php
$big   = $_POST['big'];
$small = $_POST['small'];
$n     = $_POST['n'];

// (a) Check if small string appears at start
if (stripos($big, $small) === 0)
    echo "a) '$small' appears at the start of '$big'<br>";
else
    echo "a) '$small' does NOT appear at the start of '$big'<br>";

// (b) Find position of small string in big string
$pos = stripos($big, $small);
if ($pos !== false)
    echo "b) Position of '$small' in '$big' is: $pos<br>";
else
    echo "b) '$small' not found in '$big'<br>";

// (c) Compare first n characters (case-insensitive)
if (strncasecmp($big, $small, $n) == 0)
    echo "c) First $n characters are equal (case-insensitive)";
else
    echo "c) First $n characters are NOT equal";
?>
