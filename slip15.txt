#stringops.php

	<?php
$str = $_POST['str'];

// a) First 5 words
$words = explode(" ", trim($str));
$firstFive = implode(" ", array_slice($words, 0, 5));

// b) Lowercase then Title case
$lower = strtolower($str);
$title = ucwords($lower);

// c) Pad with * on both sides
$padded = str_pad($str, strlen($str) + 4, "*", STR_PAD_BOTH);

// d) Remove leading whitespaces
$trimmed = ltrim($str);

// e) Reverse of string
$reverse = strrev($str);

echo "<h3>Results:</h3>";
echo "a) First 5 words: $firstFive <br>";
echo "b) Lowercase â†’ Title Case: $title <br>";
echo "c) Padded String: $padded <br>";
echo "d) Without leading spaces: '$trimmed' <br>";
echo "e) Reverse: $reverse";
?>
