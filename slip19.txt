#stringops.php

<?php
$big = $_POST['big']; // sentence
$small = $_POST['small']; // word
$pos = (int)$_POST['pos'];
$len = (int)$_POST['len'];
echo "<b>Original Big String:</b>
$big<br>";
echo "<b>Small String:</b>
$small<br><br>";
// a) Delete part of string
$deleted = substr_replace($big, "",
$pos, $len);
echo "a) After Deletion:
$deleted<br><br>";
// b) Insert small string into big
string
$inserted = substr_replace($big,
$small, $pos, 0);
echo "b) After Insertion:
$inserted<br><br>";
// c) Replace part of string with
small string
$replaced = substr_replace($big,
$small, $pos, $len);
echo "c) After Replacement:
$replaced<br>";
?>
